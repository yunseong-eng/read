-- Users 테이블 생성
CREATE TABLE users (
    id VARCHAR2(50) PRIMARY KEY,
    password VARCHAR2(100) NOT NULL,
    name VARCHAR2(100) NOT NULL,
    email VARCHAR2(100) NOT NULL
);

-- Board 테이블 생성
CREATE TABLE board (
    seq NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
    id VARCHAR2(50) NOT NULL,
    name VARCHAR2(100) NOT NULL,    -- 작성자 이름 추가
    phone_number VARCHAR2(20) NOT NULL, -- 핸드폰 번호 추가
    location VARCHAR2(100) NOT NULL, -- 사는 지역 추가
    logtime DATE DEFAULT SYSDATE, -- 작성날짜
    CONSTRAINT fk_board_user FOREIGN KEY (id) REFERENCES users(id)
);

SELECT seq, id, name, phone_number, location, logtime FROM board ORDER BY seq DESC;

select * from users;
select * from board;
drop table users;
